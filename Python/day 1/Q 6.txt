
#include <bits/stdc++.h>
using namespace std;
int main() {
    // Write C++ code here
    vector<int> a,b;
    int n,c;
    cin>>n;
    cout<<"Enter the values\n";
    for(int i=0;i<n;i++)
    {
        cin>>c;
        a.push_back(c);
    }
    int k;
    cout<<"Enter the value of K:";
    cin>>k;
    sort(a.begin(),a.end());
    int Mindiff=a[n-1]-a[0];
    int tempmin=a[0];
    int tempmax=a[n-1];
    for(int i=1;i<n;i++)
    {
        tempmin=min(a[0]+k,a[i]-k);
        tempmax=max(a[i-1]+k,a[n-1]-k);
        Mindiff=min(Mindiff,(tempmax-tempmin));
    }
    //cout<<tempmax<<endl<<tempmin;
    cout<<Mindiff;
    return 0;
}
