#include <bits/stdc++.h>
using namespace std;
int main() {
    // Write C++ code here
    vector<int> a;
    int n,c;
    cin>>n;
    cout<<"Enter the values\n";
    for(int i=0;i<n;i++)
    {
        cin>>c;
        a.push_back(c);
    }
    cout<<"target :";
    int target;
    cin>>target;
    sort(a.begin(),a.end());
    int i=0;
    int j=n-1;
    while(i<=j)
    {
        if(a[i]+a[j]==target)
        {
        cout<<"pair is"<<"("<<a[i]<<","<<a[j]<<")";
        break;
        }
        else if(a[i]+a[j]<target)
        i++;
        else if(a[i]+a[j]>target)
        j--;
    }
    return 0;
}
