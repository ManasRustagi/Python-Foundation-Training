#include <bits/stdc++.h>
using namespace std;
int main() {
    // Write C++ code here
    vector<int> a,b;
    int n,c;
    cin>>n;
    cout<<"Enter the values\n";
    for(int i=0;i<n;i++)
    {
        cin>>c;
        a.push_back(c);
    }
    int diff = a[1]-a[0];
    int currdiff = diff;
    int maxdiff = currdiff;
    int j,l;
    int m=0;
    int k=n;
    for(int i=1; i<n-1; i++)
    {
        diff = a[i+1]-a[i];

        if (currdiff > 0)
        {
        currdiff += diff;
        k=i+1;
        }
        else
        {
        currdiff = diff;
        m=i;
        k=i+1;
        }
        if (currdiff > maxdiff)
        {
             maxdiff = currdiff;
             j=m;
             l=k;
        }

    }
    cout<<"Maximum Difference is:"<<maxdiff<<endl;
    cout<<"Pair is ("<<a[j]<<","<<a[l]<<")";
    return 0;
}
