#include <bits/stdc++.h>
using namespace std;
int main() {
    // Write C++ code here
    vector<int> a,b;
    int n,c;
    cin>>n;
    cout<<"Enter the values\n";
    for(int i=0;i<n;i++)
    {
        cin>>c;
        a.push_back(c);
    }
    int answer=0;
    int i,j;
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            if(i!=j && a[i]==a[j]) break;
        }
        if(j==n) 
        {
            answer=a[i];
            break;
        }
    }
    cout<<"Non-repeating Element is:"<<answer;
    return 0;
}
