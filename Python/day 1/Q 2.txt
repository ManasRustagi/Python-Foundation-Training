#include <bits/stdc++.h>
using namespace std;
int main() {
    // Write C++ code here
    vector<int> a,b;
    int n,c;
    cin>>n;
    cout<<"Enter the values\n";
    for(int i=0;i<n;i++)
    {
        cin>>c;
        a.push_back(c);
    }
    for(int i=0;i<n;i++)
    {
        int product=1;
        for(int j=0;j<n;j++)
        {
            if(j!=i)
            product*=a[j];
        }
        b.push_back(product);
    }
    for(int i=0;i<n;i++)
    {
        cout<<b[i]<<",";
    }
    return 0;
}
