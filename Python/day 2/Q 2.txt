
#include <bits/stdc++.h>
using namespace std;

int Square(int &n)
{
    int i=1;
    int count=INT_MAX;
    int temp=0;
    while(i*i<=n)
    {
        int j=1;
        while(j*(i*i)<=n)
        {
            if(j*(i*i)==n)
            {
                temp=j;
                break;
            }
            else
            j++;
        }
        count=min(count,temp);
        i++;
    }
    return count;
}

int main() {
    // Write C++ code here
    int n;
    cin>>n;
    cout<<Square(n);
    return 0;
}
