
#include <bits/stdc++.h>
using namespace std;

string seq(int &num)
{
    if(num==1) return "1";
    if(num==2) return "11";
    string str="11";
    for(int i=3;i<=num;i++)
    {
        int n= str.length();
        string temp="";
        int count=1;
        for(int j=1;j<=n;j++)
        {
            if(str[j]!=str[j-1])
            {
                temp+=count+'0';
                temp+=str[j-1];
            }
            
            else
            count++;
        }
        str=temp;
    }
    return str;
}

    // Write C++ code here
    int n;
    cin>>n;
    cout<<seq(n);
    return 0;
}
